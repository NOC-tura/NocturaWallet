```typescript
import {Platform} from 'react-native';
import {API_BASE} from '../../constants/programs';
import {pinnedFetch} from '../sslPinning/pinnedFetch';
import {useSecureSettingsStore} from '../../store/zustand/secureSettingsStore';
import type {NotificationType, NotificationPayload} from './types';
import {ALL_NOTIFICATION_TYPES, NOTIFICATION_ROUTES} from './types';

const TOGGLE_KEYS: Record<NotificationType, keyof ReturnType<typeof useSecureSettingsStore.getState>> = {
  incoming_tx: 'notifIncomingTx',
  staking_reward: 'notifStakingReward',
  tx_confirmed: 'notifTxConfirmed',
  security_alert: 'notifSecurityAlert',
};

const SETTER_KEYS: Record<NotificationType, string> = {
  incoming_tx: 'setNotifIncomingTx',
  staking_reward: 'setNotifStakingReward',
  tx_confirmed: 'setNotifTxConfirmed',
  security_alert: 'setNotifSecurityAlert',
};

// Navigation ref set by app root — used for deep-linking from notification taps
let _navigationRef: {navigate: (screen: string) => void} | null = null;

export function setNavigationRef(ref: {navigate: (screen: string) => void}): void {
  _navigationRef = ref;
}

export class NotificationManager {
  private deviceToken: string | null = null;

  async requestPermission(): Promise<boolean> {
    // Stub — real implementation uses @notifee/react-native
    return true;
  }

  isEnabled(type: NotificationType): boolean {
    const state = useSecureSettingsStore.getState();
    return state[TOGGLE_KEYS[type]] as boolean;
  }

  setEnabled(type: NotificationType, enabled: boolean): void {
    const setter = SETTER_KEYS[type];
    const store = useSecureSettingsStore.getState();
    (store as Record<string, (v: boolean) => void>)[setter](enabled);
  }

  getEnabledTypes(): NotificationType[] {
    return ALL_NOTIFICATION_TYPES.filter(t => this.isEnabled(t));
  }

  async registerToken(): Promise<void> {
    const token = this.deviceToken ?? 'stub-device-token';
    const enabledTypes = this.getEnabledTypes();
    if (enabledTypes.length === 0) return;

    await pinnedFetch(`${API_BASE}/v1/notifications/register`, {
      method: 'POST',
      body: JSON.stringify({
        token,
        platform: Platform.OS as 'ios' | 'android',
        types: enabledTypes,
      }),
    });
  }

  async unregisterToken(): Promise<void> {
    const token = this.deviceToken ?? 'stub-device-token';
    await pinnedFetch(`${API_BASE}/v1/notifications/unregister`, {
      method: 'DELETE',
      body: JSON.stringify({token}),
    });
  }

  handleNotification(payload: NotificationPayload): void {
    const route = NOTIFICATION_ROUTES[payload.type];
    if (_navigationRef && route) {
      _navigationRef.navigate(route);
    }
  }
}

export const notificationManager = new NotificationManager();
```

**`src/modules/analytics/analyticsModule.ts`:**

```typescript
import {Platform} from 'react-native';
import {API_BASE} from '../../constants/programs';
import {pinnedFetch} from '../sslPinning/pinnedFetch';
import {usePublicSettingsStore} from '../../store/zustand/publicSettingsStore';
import type {AnalyticsEvent, AnalyticsPayload} from './types';

const MAX_BATCH_SIZE = 50;
const APP_VERSION = '1.0.0'; // Read from package.json or native config in production

export class AnalyticsManager {
  private queue: AnalyticsPayload[] = [];

  track(event: AnalyticsEvent): void {
    if (this.isOptedOut()) return;

    this.queue.push({
      event,
      timestamp_utc: Date.now(),
      app_version: APP_VERSION,
      platform: Platform.OS as 'ios' | 'android',
    });
  }

  async flush(): Promise<void> {
    if (this.queue.length === 0) return;

    // Process in batches of MAX_BATCH_SIZE
    while (this.queue.length > 0) {
      const batch = this.queue.splice(0, MAX_BATCH_SIZE);
      await pinnedFetch(`${API_BASE}/v1/analytics/event`, {
        method: 'POST',
        body: JSON.stringify({events: batch}),
      });
    }
  }

  isOptedOut(): boolean {
    return usePublicSettingsStore.getState().analyticsOptOut;
  }

  /** Visible for testing — returns current queue length. */
  get queueSize(): number {
    return this.queue.length;
  }

  /** Visible for testing — clears queue without sending. */
  _resetQueue(): void {
    this.queue = [];
  }
}

export const analyticsManager = new AnalyticsManager();
```

## Task 5: ChangePinScreen

### Tests (5):

1. Renders PinPad component
2. Shows "Enter current PIN" initially
3. Shows error message on wrong PIN
4. Shows "Enter new PIN" after successful verification
5. Shows "Confirm new PIN" after entering new PIN

### Implementation:

3-step state machine: `'verify' | 'new' | 'confirm'`.

```
verify: PinPad → keychainManager.verifyPin(pin)
  success → step = 'new', store nothing
  fail → show error "Current PIN is incorrect"
new: PinPad → store newPin in state
  → step = 'confirm'
confirm: PinPad → compare with newPin
  match → keychainManager.changePin(currentPin, newPin) → navigate back
  mismatch → show error "PINs do not match" (E121)
```

Uses existing `PinPad` component with `{onComplete, maxLength: 6, error, disabled}` props.

Commit: `git commit -m "feat: ChangePinScreen — 3-step verify/new/confirm flow with PinPad"`
