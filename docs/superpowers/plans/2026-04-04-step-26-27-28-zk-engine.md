```typescript
import { SystemProgram, PublicKey, TransactionInstruction } from '@solana/web3.js';
import { SHIELDED_FEES, TRANSPARENT_FEES, NOCTURA_FEE_TREASURY, API_BASE } from '../../constants/programs';
import { usePresaleStore } from '../../store/zustand/presaleStore';
import { pinnedFetch } from '../sslPinning/pinnedFetch';
import type { FeeDisplayInfo } from './types';

type ShieldedFeeType = keyof typeof SHIELDED_FEES;

const NOC_DECIMALS = 9;
const NOC_LAMPORTS = BigInt(10 ** NOC_DECIMALS);

export class FeeEngineManager {
  private cachedNocPrice: number | null = null;

  /**
   * Returns true when TGE has been activated (claimingEnabled flag in presaleStore).
   * Real on-chain check deferred; this reads the cached store state.
   */
  isTgeActive(): boolean {
    return usePresaleStore.getState().claimingEnabled;
  }

  /**
   * Returns effective fee in lamports (BigInt).
   * Pre-TGE → 0n. Zero-fee eligible → 0n.
   * Otherwise applies staking discount (if provided) using integer math.
   */
  getEffectiveFee(feeType: ShieldedFeeType, discount?: number): bigint {
    if (!this.isTgeActive()) return 0n;
    if (usePresaleStore.getState().isZeroFeeEligible) return 0n;

    const baseFee = SHIELDED_FEES[feeType];
    if (!discount || discount <= 0) return baseFee;

    const discountBps = BigInt(Math.round(discount * 100));
    return baseFee - (baseFee * discountBps / 100n);
  }

  /**
   * Builds a SystemProgram.transfer instruction that routes the transparent
   * transfer markup to NOCTURA_FEE_TREASURY.
   */
  buildTransparentFeeInstruction(sender: PublicKey): TransactionInstruction {
    return SystemProgram.transfer({
      fromPubkey: sender,
      toPubkey: new PublicKey(NOCTURA_FEE_TREASURY),
      lamports: TRANSPARENT_FEES.transferMarkup,
    });
  }

  /**
   * Formats a lamport amount into a human-readable FeeDisplayInfo.
   * 0n → "Free (until TGE)" regardless of price.
   */
  formatFeeDisplay(feeAmount: bigint, nocUsdPrice: number, discountApplied?: number): FeeDisplayInfo {
    if (feeAmount === 0n) {
      return {
        amount: '0',
        displayText: 'Free (until TGE)',
        discountApplied,
      };
    }

    const nocAmount = Number(feeAmount) / Number(NOC_LAMPORTS);
    const usdAmount = nocAmount * nocUsdPrice;

    const nocStr = nocAmount.toFixed(4).replace(/\.?0+$/, '');
    const usdStr = usdAmount.toFixed(6).replace(/\.?0+$/, '') || '0';

    return {
      amount: feeAmount.toString(),
      displayText: `${nocStr} NOC ≈ $${usdStr}`,
      discountApplied,
    };
  }

  /**
   * Fetches current NOC/USD price from the Noctura price API.
   * Returns cached value on subsequent calls within the same instance.
   */
  async getNocUsdPrice(): Promise<number> {
    if (this.cachedNocPrice !== null) return this.cachedNocPrice;

    try {
      const res = await pinnedFetch(`${API_BASE}/v1/price/noc`);
      const data = await res.json() as { price: number };
      this.cachedNocPrice = data.price;
      return data.price;
    } catch {
      // Fallback: return 0 if price fetch fails; caller handles gracefully
      return 0;
    }
  }
}

export const feeEngine = new FeeEngineManager();
```

```typescript
import { mmkvPublic } from '../../store/mmkv/instances';
import { MMKV_KEYS } from '../../constants/mmkvKeys';
import { pinnedFetch } from '../sslPinning/pinnedFetch';
import { API_BASE } from '../../constants/programs';
import type { MerkleState, AnchorResponse, MerkleUpdate } from './types';

export interface AnchorVerifyResult {
  synced: boolean;
  remoteRoot: string;
  localRoot: string | null;
}

export class MerkleManager {
  private getLocalState(): MerkleState | null {
    const raw = mmkvPublic.getString(MMKV_KEYS.SHIELDED_MERKLE_STATE);
    if (!raw) return null;
    try {
      return JSON.parse(raw) as MerkleState;
    } catch {
      return null;
    }
  }

  private saveLocalState(state: MerkleState): void {
    mmkvPublic.set(MMKV_KEYS.SHIELDED_MERKLE_STATE, JSON.stringify(state));
  }

  /**
   * Fetches the on-chain Merkle anchor and compares to local stored root.
   * Returns synced=true if roots match, false otherwise.
   */
  async verifyAnchor(): Promise<AnchorVerifyResult> {
    const res = await pinnedFetch(`${API_BASE}/v1/merkle/anchor`);
    const anchor = await res.json() as AnchorResponse;

    const localState = this.getLocalState();
    const localRoot = localState?.root ?? null;
    const synced = localRoot === anchor.root;

    return {
      synced,
      remoteRoot: anchor.root,
      localRoot,
    };
  }

  /**
   * Fetches leaves [fromLeaf, fromLeaf+count) from the API and persists each
   * to MMKV under SHIELDED_MERKLE_LEAVES_PREFIX + index.
   * Returns the new total leaf count.
   */
  async syncFromLeaf(fromLeaf: number, toLeaf: number): Promise<number> {
    const res = await pinnedFetch(
      `${API_BASE}/v1/merkle/updates?fromLeaf=${fromLeaf}&toLeaf=${toLeaf}`,
    );
    const update = await res.json() as MerkleUpdate;

    update.leaves.forEach((leaf, i) => {
      mmkvPublic.set(MMKV_KEYS.SHIELDED_MERKLE_LEAVES_PREFIX + (fromLeaf + i), leaf);
    });

    const newLeafCount = fromLeaf + update.leaves.length;

    const existing = this.getLocalState();
    this.saveLocalState({
      root: existing?.root ?? '',
      leafCount: newLeafCount,
      lastSyncedAt: Date.now(),
    });

    return newLeafCount;
  }

  /**
   * Simplified local root computation: encodes the stored root string as UTF-8 bytes.
   * Real implementation deferred (needs Poseidon tree over all leaves).
   */
  computeLocalRoot(): Uint8Array {
    const state = this.getLocalState();
    const rootStr = state?.root ?? '';
    // Encode stored root hex string as UTF-8 bytes (simplified — not real Poseidon)
    const encoder = new TextEncoder();
    return encoder.encode(rootStr);
  }

  /**
   * Clears all stored Merkle keys (state + all leaves) then re-syncs from leaf 0.
   */
  async fullResync(): Promise<void> {
    // Clear state
    mmkvPublic.remove(MMKV_KEYS.SHIELDED_MERKLE_STATE);

    // Clear all leaf keys
    const allKeys = mmkvPublic.getAllKeys();
    for (const key of allKeys) {
      if (key.startsWith(MMKV_KEYS.SHIELDED_MERKLE_LEAVES_PREFIX)) {
        mmkvPublic.remove(key);
      }
    }

    // Re-sync from leaf 0 — fetch all available leaves
    // API with toLeaf=0 signals "give me everything from 0"
    await this.syncFromLeaf(0, 0);
  }

  /**
   * Stub: scanning for notes requires ZK circuit-compatible decryption (sk_view).
   * Returns empty array until ZK circuit integration is complete.
   */
  async scanForNotes(): Promise<unknown[]> {
    return [];
  }
}

export const merkleManager = new MerkleManager();
```

```typescript
import { mmkvPublic } from '../../store/mmkv/instances';
import { MMKV_KEYS } from '../../constants/mmkvKeys';
import type { ProofQueueEntry } from './types';

const MAX_ATTEMPTS = 3;
const STALE_THRESHOLD_MS = 10 * 60 * 1000; // 10 minutes

function generateId(): string {
  return Date.now().toString(36) + Math.random().toString(36).slice(2);
}

export class ProofQueue {
  private readAll(): ProofQueueEntry[] {
    const raw = mmkvPublic.getString(MMKV_KEYS.PROOF_QUEUE);
    if (!raw) return [];
    try {
      return JSON.parse(raw) as ProofQueueEntry[];
    } catch {
      return [];
    }
  }

  private writeAll(entries: ProofQueueEntry[]): void {
    mmkvPublic.set(MMKV_KEYS.PROOF_QUEUE, JSON.stringify(entries));
  }

  addToQueue(
    params: Omit<ProofQueueEntry, 'id' | 'attempts' | 'createdAt' | 'status'>,
  ): ProofQueueEntry {
    const entry: ProofQueueEntry = {
      ...params,
      id: generateId(),
      attempts: 0,
      createdAt: Date.now(),
      status: 'pending',
    };
    const all = this.readAll();
    all.push(entry);
    this.writeAll(all);
    return entry;
  }

  getPendingEntries(): ProofQueueEntry[] {
    return this.readAll().filter(
      e => e.status === 'pending' && e.attempts < MAX_ATTEMPTS,
    );
  }

  updateEntry(
    id: string,
    updates: Partial<ProofQueueEntry>,
  ): ProofQueueEntry | null {
    const all = this.readAll();
    const index = all.findIndex(e => e.id === id);
    if (index === -1) return null;
    const updated = { ...all[index], ...updates };
    all[index] = updated;
    this.writeAll(all);
    return updated;
  }

  /**
   * Marks any pending entries older than STALE_THRESHOLD_MS as failed.
   */
  cleanupStale(): void {
    const now = Date.now();
    const all = this.readAll().map(entry => {
      if (entry.status === 'pending' && now - entry.createdAt > STALE_THRESHOLD_MS) {
        return { ...entry, status: 'failed' as const };
      }
      return entry;
    });
    this.writeAll(all);
  }
}

export const proofQueue = new ProofQueue();
```

```typescript
import { pinnedFetch } from '../sslPinning/pinnedFetch';
import { API_BASE } from '../../constants/programs';
import { ProofQueue } from './proofQueue';
import { LocalProver } from './localProver';
import type { ProofResult, DepositProofParams, TransferProofParams, WithdrawProofParams } from './types';

type CircuitType = 'deposit' | 'transfer' | 'withdraw';

const HOSTED_TIMEOUT_MS = 8_000;

function containsSpendKey(params: Record<string, unknown>): boolean {
  return Object.keys(params).some(k => k.toLowerCase().includes('spend'));
}

function zeroizeWitness(witness: Uint8Array | undefined): void {
  if (witness) {
    witness.fill(0);
  }
}

export class ZKProverManager {
  private queue = new ProofQueue();
  private local = new LocalProver();

  getProverMode(): 'hosted' | 'local' {
    return 'hosted';
  }

  private async proveHosted(
    circuitType: CircuitType,
    params: Record<string, unknown>,
  ): Promise<ProofResult> {
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), HOSTED_TIMEOUT_MS);

    try {
      const res = await pinnedFetch(`${API_BASE}/v1/prove/${circuitType}`, {
        method: 'POST',
        body: JSON.stringify(params),
      });
      const data = await res.json() as ProofResult;
      return data;
    } finally {
      clearTimeout(timer);
    }
  }

  private async generateProof(
    circuitType: CircuitType,
    params: Record<string, unknown>,
    witness?: Uint8Array,
  ): Promise<ProofResult> {
    if (containsSpendKey(params)) {
      throw new Error('sk_spend must never be included in proof params');
    }

    try {
      const result = await this.proveHosted(circuitType, params);
      zeroizeWitness(witness);
      return result;
    } catch (hostedError) {
      zeroizeWitness(witness);

      // Check if local prover can handle it
      const viability = this.local.checkViability();
      if (viability.supported) {
        // Local proving path (future integration)
        // Falls through to queue for now
      }

      // Add to proof queue for retry
      this.queue.addToQueue({ circuitType, params });
      throw new Error(`Proof generation queued for retry: ${(hostedError as Error).message}`);
    }
  }

  async generateDepositProof(
    params: DepositProofParams & Record<string, unknown>,
    witness?: Uint8Array,
  ): Promise<ProofResult> {
    return this.generateProof('deposit', params, witness);
  }

  async generateTransferProof(
    params: TransferProofParams & Record<string, unknown>,
    witness?: Uint8Array,
  ): Promise<ProofResult> {
    return this.generateProof('transfer', params, witness);
  }

  async generateWithdrawProof(
    params: WithdrawProofParams & Record<string, unknown>,
    witness?: Uint8Array,
  ): Promise<ProofResult> {
    return this.generateProof('withdraw', params, witness);
  }
}

export const zkProver = new ZKProverManager();
```
