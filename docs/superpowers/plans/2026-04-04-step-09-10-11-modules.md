# Step 9–10–11: Geo-Fence, Token Module, Device Integrity

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build three independent modules: (1) geo-fence with 3-tier soft-block policy, VPN detection, and cached restricted lists, (2) token management with 3-tier trust model, Jupiter token list integration, scam flag checking, and rent reclaim for empty accounts, (3) device integrity checks (jailbreak/root detection) and screen security native bridge stubs.

**Architecture:** Each module is fully independent — no cross-dependencies between them. GeoFenceModule fetches a restricted country list from the API with 6h MMKV cache and falls back to a bundled list. TokenModule wraps token account queries with trust tier classification and scam flag checking. DeviceIntegrityModule checks jailbreak/root status at app start and stores the result in MMKV for UI consumers. ScreenSecurityModule provides a NativeModules bridge for FLAG_SECURE (Android) and isCaptured detection (iOS) — stubs until native code is built.

**Tech Stack:** react-native-config, react-native-mmkv, pinnedFetch (SSL-pinned API client), @solana/web3.js queries

```typescript
import {pinnedFetch} from '../sslPinning/pinnedFetch';
import {API_BASE} from '../../constants/programs';
import {MMKV_KEYS} from '../../constants/mmkvKeys';
import {mmkvPublic} from '../../store/mmkv/instances';
import {BUNDLED_RESTRICTED_LIST, type RestrictedCountry} from './restrictedList';

export interface JurisdictionResult {
  action: 'allow' | 'warn' | 'block';
  countryCode: string;
  reason?: 'restricted' | 'sanctioned' | 'ambiguous' | 'vpn_detected';
  transparentAllowed: true;
  message?: string;
}

interface GeoCheckResponse {
  countryCode: string;
  isVpn: boolean;
}

const RESTRICTED_LIST_CACHE_MS = 6 * 60 * 60 * 1000; // 6 hours

/**
 * Geo-fence module with 3-tier soft-block policy.
 *
 * allow:  unrestricted jurisdiction — full access
 * warn:   restricted region or VPN detected — shielded accessible after acknowledgment
 * block:  OFAC sanctioned ONLY — shielded disabled, transparent works normally
 *
 * ⚠️ SOFT-BLOCK: false positives must NOT brick the wallet.
 * ⚠️ Fail-safe: if detection fails or is ambiguous → 'warn' (NEVER silent block).
 */
export class GeoFenceManager {
  private kycCountry: string | null = null;

  /**
   * Check user's jurisdiction. Called on app open + shielded mode toggle.
   */
  async checkJurisdiction(): Promise<JurisdictionResult> {
    try {
      // If KYC country is set, use it instead of IP detection
      if (this.kycCountry) {
        return this.classifyCountry(this.kycCountry);
      }

      const response = await pinnedFetch(`${API_BASE}/v1/geo/check`);
      const data = (await response.json()) as GeoCheckResponse;

      // VPN detection
      if (data.isVpn) {
        return {
          action: 'warn',
          countryCode: data.countryCode,
          reason: 'vpn_detected',
          transparentAllowed: true,
          message: 'VPN detected. Set your country manually in Settings for accurate access.',
        };
      }

      return this.classifyCountry(data.countryCode);
    } catch {
      // Fail-safe: API unreachable → warn, never silent block
      return {
        action: 'warn',
        countryCode: 'unknown',
        reason: 'ambiguous',
        transparentAllowed: true,
        message: 'Unable to verify your region. Some features may be restricted.',
      };
    }
  }

  /**
   * Override IP-based detection with KYC-verified country.
   */
  setKycCountry(countryCode: string): void {
    this.kycCountry = countryCode;
    mmkvPublic.set(MMKV_KEYS.GEO_KYC_COUNTRY, countryCode);
  }

  /**
   * Classify a country code against the restricted list.
   */
  private classifyCountry(countryCode: string): JurisdictionResult {
    const list = this.getRestrictedList();
    const entry = list.find(c => c.code === countryCode);

    if (!entry) {
      return {
        action: 'allow',
        countryCode,
        transparentAllowed: true,
      };
    }

    if (entry.category === 'sanctioned') {
      return {
        action: 'block',
        countryCode,
        reason: 'sanctioned',
        transparentAllowed: true,
        message: 'Privacy mode is not available in your region due to regulatory restrictions. Transparent mode is fully functional.',
      };
    }

    return {
      action: 'warn',
      countryCode,
      reason: 'restricted',
      transparentAllowed: true,
      message: 'Your region may have restrictions on privacy tools. By continuing you accept responsibility for local compliance.',
    };
  }

  /**
   * Get restricted list from MMKV cache or bundled fallback.
   */
  private getRestrictedList(): RestrictedCountry[] {
    const cachedJson = mmkvPublic.getString(MMKV_KEYS.GEO_RESTRICTED_LIST);
    const cachedAt = mmkvPublic.getString(MMKV_KEYS.GEO_RESTRICTED_LIST_AT);

    if (cachedJson && cachedAt) {
      const age = Date.now() - Number(cachedAt);
      if (age < RESTRICTED_LIST_CACHE_MS) {
        try {
          return JSON.parse(cachedJson) as RestrictedCountry[];
        } catch {
          // Corrupted cache — fall through to bundled list
        }
      }
    }

    // Trigger async refresh in background (fire and forget)
    this.refreshRestrictedList();

    return BUNDLED_RESTRICTED_LIST;
  }

  /**
   * Fetch fresh restricted list from API and cache in MMKV.
   */
  private async refreshRestrictedList(): Promise<void> {
    try {
      const response = await pinnedFetch(`${API_BASE}/v1/geo/restricted-list`);
      const data = (await response.json()) as {countries: RestrictedCountry[]};
      mmkvPublic.set(MMKV_KEYS.GEO_RESTRICTED_LIST, JSON.stringify(data.countries));
      mmkvPublic.set(MMKV_KEYS.GEO_RESTRICTED_LIST_AT, String(Date.now()));
    } catch {
      // API unavailable — bundled list remains in use
    }
  }
}
```

```typescript
import {CORE_MINTS, CORE_TOKENS} from './coreTokens';
import {NOC_MINT} from '../../constants/programs';
import {pinnedFetch} from '../sslPinning/pinnedFetch';
import {API_BASE} from '../../constants/programs';

type TrustTier = 'core' | 'verified' | 'unknown';

interface SimpleAccount {
  mint: string;
  balance: string;
  address: string;
}

/**
 * Token management module.
 *
 * 3-tier trust model:
 *   core:     NOC, SOL, USDC, USDT — hardcoded, always shown
 *   verified: Jupiter "strict" list (verified badge)
 *   unknown:  other SPL tokens — ⚠️ "Unverified token" warning
 *
 * Scam flag: GET API_BASE/v1/tokens/flagged?mint=... (cache 1h)
 * If flagged: ⛔ icon + disable send (user can explicit override)
 */
export class TokenManager {
  /**
   * Classify a token mint into a trust tier.
   * Core tokens are hardcoded. Verified tokens come from Jupiter strict list (to be integrated).
   * Everything else is unknown.
   */
  classifyToken(mint: string): TrustTier {
    if (CORE_MINTS.has(mint)) return 'core';
    // Jupiter verified list integration will be added when token list fetching is built
    return 'unknown';
  }

  /**
   * Check if a token is flagged as a scam.
   * Returns false on API error (fail-safe — don't block legitimate tokens).
   */
  async checkScamFlag(mint: string): Promise<boolean> {
    try {
      const response = await pinnedFetch(
        `${API_BASE}/v1/tokens/flagged?mint=${encodeURIComponent(mint)}`,
      );
      const data = (await response.json()) as {flagged: boolean; reason?: string};
      return data.flagged;
    } catch {
      return false;
    }
  }

  /**
   * Get empty token accounts eligible for rent reclaim.
   * Excludes core tokens (NOC, SOL, USDC, USDT) — NEVER close these.
   */
  getEmptyAccountsForCleanup(accounts: SimpleAccount[]): SimpleAccount[] {
    return accounts.filter(
      account => account.balance === '0' && !CORE_MINTS.has(account.mint),
    );
  }

  /**
   * Sort tokens with NOC pinned at top, then core tokens, then by symbol.
   */
  sortTokens<T extends {mint: string; symbol: string}>(tokens: T[]): T[] {
    return [...tokens].sort((a, b) => {
      // NOC always first
      if (a.mint === NOC_MINT) return -1;
      if (b.mint === NOC_MINT) return 1;
      // Core tokens next
      const aCore = CORE_MINTS.has(a.mint);
      const bCore = CORE_MINTS.has(b.mint);
      if (aCore && !bCore) return -1;
      if (!aCore && bCore) return 1;
      // Alphabetical
      return a.symbol.localeCompare(b.symbol);
    });
  }
}
```

```typescript
import JailMonkey from 'jail-monkey';
import {mmkvPublic} from '../../store/mmkv/instances';
import {MMKV_KEYS} from '../../constants/mmkvKeys';

/**
 * Device integrity module — jailbreak/root detection.
 *
 * Policy: WARN, NOT BLOCK.
 *   - Jailbroken/rooted device = Secure Enclave/Keystore potentially compromised
 *   - But false positives (developer mode, custom ROM) would brick the wallet
 *   - So we warn persistently and degrade security (disable biometric on jailbreak)
 *
 * Flow:
 *   1. App start → checkDeviceIntegrity()
 *   2. If detected → MMKV SECURITY_JAILBREAK_DETECTED = true
 *   3. Continue normally (no block)
 *   4. UI consumers read isCompromised() for:
 *      - UnlockScreen: warning below PIN pad
 *      - Settings → Security: persistent warning card
 *      - Shielded mode toggle: require explicit acknowledgment
 */
export class DeviceIntegrityManager {
  /**
   * Run integrity check. Call on app startup.
   * Sets MMKV flag if compromised — does NOT block.
   */
  checkDeviceIntegrity(): void {
    try {
      const jailbroken = JailMonkey.isJailBroken();
      if (jailbroken) {
        mmkvPublic.set(MMKV_KEYS.SECURITY_JAILBREAK_DETECTED, 'true');
      }
    } catch {
      // jail-monkey not available (e.g., some Android ROMs) — fail open
    }
  }

  /**
   * Check if device is compromised (reads from MMKV).
   * Returns true if jailbreak/root was ever detected.
   */
  isCompromised(): boolean {
    return mmkvPublic.getString(MMKV_KEYS.SECURITY_JAILBREAK_DETECTED) === 'true';
  }

  /**
   * Get a human-readable warning message for UI display.
   * Returns null if device is not compromised.
   */
  getWarningMessage(): string | null {
    if (!this.isCompromised()) return null;
    return 'Your device appears to be compromised. Your wallet keys may be at risk. Biometric authentication has been disabled for security.';
  }
}
```
